   - Default behavior: return placeholder (current behavior for testing)

```python
from typing import Callable, Optional
import json

# MCP invocation callback - set by caller that has MCP access
_mcp_invoke_callback: Optional[Callable[[str, dict], str]] = None

def set_mcp_invoke_callback(callback: Callable[[str, dict], str]) -> None:
    """Set callback for MCP tool invocation. Callback takes (tool_name, params) and returns response string."""
    global _mcp_invoke_callback
    _mcp_invoke_callback = callback

def _invoke_mcp(tool_name: str, params: dict) -> Optional[str]:
    """Invoke MCP tool if callback is set, otherwise return None."""
    if _mcp_invoke_callback is None:
        return None
    try:
        return _mcp_invoke_callback(tool_name, params)
    except Exception as e:
        logger.warning(f"MCP invocation failed for {tool_name}: {e}")
        return None
```

```python
def _parse_markdown_table(response: str) -> list[dict]:
    """Parse markdown table from MCP response into list of dicts."""
    lines = response.strip().split('\n')
    if len(lines) < 3:  # Need header, separator, at least one row
        return []

    # Extract header row
    headers = [h.strip() for h in lines[0].split('|') if h.strip()]

    # Parse data rows (skip separator line at index 1)
    rows = []
    for line in lines[2:]:
        if not line.strip() or line.startswith('---'):
            continue
        values = [v.strip() for v in line.split('|') if v.strip()]
        if len(values) == len(headers):
            rows.append(dict(zip(headers, values)))
    return rows

def _parse_currency(value: str) -> float:
    """Parse currency string like '$1.2M' or '$500K' to float."""
    if not value:
        return 0.0
    value = value.replace('$', '').replace(',', '').strip()
    multiplier = 1
    if value.endswith('K'):
        multiplier = 1_000
        value = value[:-1]
    elif value.endswith('M'):
        multiplier = 1_000_000
        value = value[:-1]
    elif value.endswith('B'):
        multiplier = 1_000_000_000
        value = value[:-1]
    try:
        return float(value) * multiplier
    except ValueError:
        return 0.0

def _calculate_flow_confidence(rows: list[dict], volume_threshold: float = 100_000) -> int:
    """Calculate confidence score based on flow data quality."""
    if not rows:
        return 0
    # More rows = more data points = higher confidence
    row_score = min(len(rows) * 5, 30)  # Up to 30 points for data volume

    # Check for recent data (last 24h)
    # If we have data, base confidence on it
    total_volume = sum(_parse_currency(row.get('Balance USD', '0')) for row in rows)
    volume_score = min(int(total_volume / volume_threshold), 50)  # Up to 50 points for volume

    # Data freshness score (always give some credit if we have data)
    freshness_score = 20 if rows else 0

    return min(row_score + volume_score + freshness_score, 100)
```

These helpers enable the fetch functions to convert MCP markdown responses into structured signal data with computed confidence scores.
